Under The Hood

Open source. Simple by design.

No complex frameworks. No vendor lock-in. Python + SQLite + your preferred chat API. Every feature is a module with three methods.

modules/your_feature/module.py 
class YourFeature(BaseModule):

    def can_handle(self, message) -> bool:
        # Does this message belong to your module?
        return "keyword" in message.text.lower()

    def handle(self, message) -> Response:
        # Process it and return a reply
        self.db.save(message)
        return Response(reply="Done!")

    def get_scheduled_jobs(self) -> list:
        # Automated messages (reminders, reports)
        return [ScheduledJob(cron="0 22 * * *", func=self.daily_report)]
